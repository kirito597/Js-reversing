环境检测原理

网站使用环境检测来识别自动化工具、爬虫或异常环境，主要原理包括：

  • 检测浏览器特有属性和方法
  • 检查JavaScript引擎的特殊行为
  • 验证DOM和BOM对象的完整性
  • 检测自动化工具特征（如webdriver标志）
  • 验证时间戳和性能指标
// 典型的环境检测代码示例
if (navigator.webdriver) {
    console.log("检测到自动化工具");
    // 可能阻止访问或返回虚假数据
}

if (window.outerWidth === 0 && window.outerHeight === 0) {
    console.log("可能是无头浏览器");
}

if (!window.chrome || !navigator.languages) {
    console.log("环境不完整");
}

常见环境检测点

以下是一些网站常用的环境检测点，逆向时需要特别注意：

1. Navigator 对象检测

  • userAgent, platform, language
  • webdriver 标志 (常见于Puppeteer/Playwright)
  • plugins, mimeTypes
  • hardwareConcurrency, deviceMemory

2. Window 和 Screen 检测

  • outerWidth, outerHeight, innerWidth, innerHeight
  • screenX, screenY, availWidth, availHeight
  • devicePixelRatio

3. 高级检测

  • Canvas 指纹检测
  • WebGL 渲染器检测
  • 音频上下文指纹
  • 字体枚举检测

补环境技巧

以下是一些常用的补环境技巧：

1. 属性覆盖

// 覆盖只读属性
Object.defineProperty(navigator, 'webdriver', {
    get: () => false,
    configurable: true
});

// 覆盖函数
navigator.__defineGetter__('languages', () => ['zh-CN', 'zh', 'en']);

2. 代理拦截

// 使用Proxy拦截对象访问
window = new Proxy(window, {
    get: function(target, key) {
        if (key === 'outerWidth') {
            return 1920; // 返回合理值
        }
        return target[key];
    }
});

3. 环境预填充

// 预先填充缺失的对象和方法
if (!window.chrome) {
    window.chrome = {
        runtime: {},
        // 其他chrome特有属性
    };
}

实战示例

以下是一个完整的补环境示例，模拟真实浏览器环境：

// 补navigator对象
function patchNavigator() {
    const originalNavigator = navigator;

    // 覆盖webdriver属性
    Object.defineProperty(navigator, 'webdriver', {
        get: () => false,
        configurable: true
    });

    // 添加缺失的属性
    if (!navigator.languages) {
        navigator.languages = ['zh-CN', 'zh', 'en'];
    }

    // 保护navigator对象不被修改
    return new Proxy(originalNavigator, {
        set: function(target, key, value) {
            // 阻止修改关键属性
            if (['webdriver', 'languages'].includes(key)) {
                return true; // 静默失败
            }
            target[key] = value;
            return true;
        }
    });
}

// 补window对象
function patchWindow() {
    // 确保窗口尺寸合理
    if (window.outerWidth === 0) {
        window.outerWidth = 1920;
        window.outerHeight = 1080;
    }

    // 添加缺失的chrome对象
    if (!window.chrome) {
        window.chrome = {
            runtime: {},
            loadTimes: function() {},
            csi: function() {},
            app: {
                isInstalled: false,
                InstallState: 'disabled',
                RunningState: 'stopped'
            }
        };
    }
}

// 执行补环境
patchNavigator();
patchWindow();
